什么是undefined数据类型？

//1、一个变量声明了，但是没有赋值，值默认为
undefined var b;
console.log(b); //b就是一个undefined类型的值

//2、一个变量声明了，并且赋值了一个undefined的值
var c= undefined;
console.log(c); //c也是一个undefined类型的值

//3、一个对象中，获取某个不存在的属性，值也是undefined
var d={};
console.log(d.age); //由于d对象没有age属性，值：undefined
